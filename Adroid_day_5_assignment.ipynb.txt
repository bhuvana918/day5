        "# Load necessary libraries\n",
        "library(tidyverse)\n",
        "library(reshape2)\n",
        "library(ggplot2)\n",
        "\n",
        "# Step 1: Create a random dataset of 100 rows and 30 columns with values between 1 and 200\n",
        "set.seed(123)  # for reproducibility\n",
        "dataset <- as.data.frame(matrix(sample(1:200, 100*30, replace = TRUE), nrow = 100, ncol = 30))\n",
        "\n",
        "# Step 2: Replace all values between [10, 60] with NA and count rows with missing values\n",
        "dataset[dataset >= 10 & dataset <= 60] <- NA\n",
        "rows_with_na <- sum(apply(dataset, 1, function(x) any(is.na(x))))\n",
        "print(paste(\"Number of rows with missing values:\", rows_with_na))\n",
        "\n",
        "# Step 3: Replace all NA values with the average of the column value\n",
        "for(i in 1:ncol(dataset)) {\n",
        "  dataset[is.na(dataset[, i]), i] <- mean(dataset[, i], na.rm = TRUE)\n",
        "}\n",
        "\n",
        "# Step 4: Find the Pearson correlation among all columns and plot heatmap\n",
        "cor_matrix <- cor(dataset)\n",
        "melted_cor_matrix <- melt(cor_matrix)\n",
        "\n",
        "ggplot(data = melted_cor_matrix, aes(x=Var1, y=Var2, fill=value)) +\n",
        "  geom_tile() +\n",
        "  scale_fill_gradient2(low = \"blue\", high = \"red\", mid = \"white\",\n",
        "                       midpoint = 0, limit = c(-1,1), space = \"Lab\",\n",
        "                       name=\"Pearson\\nCorrelation\") +\n",
        "  theme_minimal() +\n",
        "  theme(axis.text.x = element_text(angle = 45, vjust = 1,\n",
        "                                   size = 12, hjust = 1)) +\n",
        "  coord_fixed()\n",
        "\n",
        "# Select columns with correlation ≤ 0.7\n",
        "selected_columns <- names(which(apply(cor_matrix, 2, function(x) any(abs(x) <= 0.7))))\n",
        "print(\"Columns with correlation <= 0.7:\")\n",
        "print(selected_columns)\n",
        "\n",
        "# Step 5: Normalize all values in the dataset between 0 and 10\n",
        "normalize <- function(x) {\n",
        "  return ((x - min(x)) / (max(x) - min(x)) * 10)\n",
        "}\n",
        "normalized_dataset <- as.data.frame(lapply(dataset, normalize))\n",
        "\n",
        "# Step 6: Replace all values with 1 if value ≤ 0.5, else with 0\n",
        "final_dataset <- as.data.frame(lapply(normalized_dataset, function(x) ifelse(x <= 0.5, 1, 0)))\n",
        "\n",
        "# Display the final dataset\n",
        "print(\"Final dataset after replacing values with 1 or 0:\")\n",
        "print(head(final_dataset))\n"
